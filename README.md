# dg
dynamic geometry software

DGS (dynamic geometry software)

API

dg.geom namespace

geom type ids: 
	1 -> point, 
	2 -> vector,
	3 -> midpoint,
	4 -> line,
	5 -> segment,
	6 -> perpendicular line,
	7 -> perpendicular bisector,
	8 -> parallel line,
	9 -> angle bisector,
	10 -> circle,
	11 -> tangent,
	12 -> polygon,
	13 -> function,
	14 -> intersection
	15 -> angle
